<script>
    const database = {
        "أطعمة": ["بيتزا", "شاورما", "كبسة", "برجر", "سوشي", "ورق عنب", "ملوخية", "تورتة", "باستا", "فلافل", "مندي", "كنافة"],
        "مهن": ["طيار", "طبيب", "مهندس", "نجار", "محامي", "مبرمج", "حلاق", "شرطي", "معلم", "طباخ", "فنان", "مزارع"],
        "حيوانات": ["أسد", "فيل", "زرافة", "قطة", "صقر", "جمل", "نمر", "ذئب", "ثعلب", "غزال", "تمساح", "قرد"]
    };

    let players = [];
    let scores = {};
    let selectedTopic = "";
    let selectedCategory = "";
    let imposterIdx = -1;
    let currentIdx = 0;
    let qIdx = 0;

    function addPlayer() {
        const input = document.getElementById('player-input');
        if(input.value.trim()){
            let name = input.value.trim();
            if(players.includes(name)) return alert("الاسم موجود!");
            players.push(name);
            scores[name] = scores[name] || 0;
            input.value = "";
            updatePlayersUI();
        }
    }

    function updatePlayersUI() {
        const div = document.getElementById('players-list-ui');
        div.innerHTML = players.map((p, i) => `
            <span class="player-tag">${p} <b style="color:red;cursor:pointer" onclick="removePlayer(${i})">×</b></span>
        `).join('');
    }

    function removePlayer(i) {
        players.splice(i, 1);
        updatePlayersUI();
    }

    function goToCategorySelection() {
        if(players.length < 3) return alert("تحتاج 3 لاعبين على الأقل!");
        showScreen('category-screen');
    }

    function startGame(category) {
        selectedCategory = category;
        const list = database[category];
        selectedTopic = list[Math.floor(Math.random() * list.length)];
        imposterIdx = Math.floor(Math.random() * players.length);
        currentIdx = 0;
        showScreen('reveal-screen');
        setupReveal();
    }

    function setupReveal() {
        document.getElementById('turn-name').innerText = `مرر الجهاز لـ: ${players[currentIdx]}`;
        document.getElementById('tap-msg').classList.remove('hidden');
        document.getElementById('secret-text').classList.add('hidden');
        document.getElementById('show-btn').classList.remove('hidden');
        document.getElementById('next-player-btn').classList.add('hidden');
        
        if(currentIdx === imposterIdx) {
            document.getElementById('secret-text').innerText = "🚫 أنت برا السالفة!";
            document.getElementById('secret-text').style.color = "var(--accent)";
        } else {
            document.getElementById('secret-text').innerText = `السالفة هي: ${selectedTopic}`;
            document.getElementById('secret-text').style.color = "var(--primary)";
        }
    }

    function revealInfo() {
        document.getElementById('tap-msg').classList.add('hidden');
        document.getElementById('secret-text').classList.remove('hidden');
        document.getElementById('show-btn').classList.add('hidden');
        document.getElementById('next-player-btn').classList.remove('hidden');
    }

    function nextPlayerReveal() {
        currentIdx++;
        if(currentIdx < players.length) setupReveal();
        else startQuestions();
    }

    function startQuestions() {
        showScreen('question-screen');
        qIdx = 0;
        document.getElementById('q-options').classList.add('hidden');
        document.getElementById('q-next-btn').classList.remove('hidden');
        cycleQuestions();
    }

    function cycleQuestions() {
        if(qIdx < players.length) {
            let asker = players[qIdx];
            let target = players[(qIdx + 1) % players.length];
            document.getElementById('question-content').innerHTML = `🎤 <span style="color:var(--primary)">${asker}</span> اسأل <span style="color:var(--accent)">${target}</span>`;
            qIdx++;
        } else {
            document.getElementById('question-content').innerText = "✅ انتهت الأسئلة! حان وقت كشف المستور.";
            document.getElementById('q-next-btn').classList.add('hidden');
            document.getElementById('q-options').classList.remove('hidden');
        }
    }

    function startVoting() {
        showScreen('vote-screen');
        currentIdx = 0;
        processVote();
    }

    function processVote() {
        let activeVoter = players[currentIdx];
        document.getElementById('voter-title').innerText = `🗳️ دور ${activeVoter} في التصويت:`;
        const container = document.getElementById('vote-buttons');
        container.innerHTML = players
            .filter(p => p !== activeVoter) 
            .map(p => `<button class="btn" onclick="handleVote('${p}')">${p}</button>`)
            .join('');
    }

    function handleVote(votedName) {
        if(votedName === players[imposterIdx]) scores[players[currentIdx]]++;
        currentIdx++;
        if(currentIdx < players.length) processVote();
        else revealImposter();
    }

    function revealImposter() {
        showScreen('result-screen');
        let imposterName = players[imposterIdx];
        document.getElementById('imposter-name-final').innerText = imposterName;
        document.getElementById('imposter-quiz').classList.add('hidden');
        document.getElementById('to-final-btn').classList.add('hidden');
        
        setTimeout(() => {
            document.getElementById('imposter-quiz').classList.remove('hidden');
            document.getElementById('imposter-question-text').innerText = `يا ${imposterName}، إيش هي السالفة؟ (الفئة: ${selectedCategory})`;
            
            let categoryWords = database[selectedCategory].filter(w => w !== selectedTopic);
            let wrongOptions = categoryWords.sort(() => Math.random() - 0.5).slice(0, 4);
            let finalOptions = [selectedTopic, ...wrongOptions].sort(() => Math.random() - 0.5);

            document.getElementById('imposter-options').innerHTML = finalOptions.map(o => 
                `<button class="btn" style="background:#f1f2f6; color:#333; border:1px solid #ccc" onclick="checkImposter('${o}')">${o}</button>`
            ).join('');
        }, 1500);
    }

    function checkImposter(guess) {
        if(guess === selectedTopic) {
            scores[players[imposterIdx]] += 2; // نقطتين إذا عرف السالفة
            alert("👏 ذكي! عرفت السالفة وكسبت نقطتين.");
        } else {
            alert(`❌ للأسف! السالفة كانت: ${selectedTopic}`);
        }
        document.getElementById('to-final-btn').classList.remove('hidden');
    }

    function showScores() {
        showScreen('final-screen');
        const res = document.getElementById('score-results');
        res.innerHTML = players.sort((a,b) => scores[b] - scores[a]).map(p => `<p>${p}: <b>${scores[p]} نقطة</b></p>`).join('');
        updateHighscore();
    }

    function updateHighscore() {
        let max = -1; let winner = "لا يوجد";
        for(let p in scores) { if(scores[p] > max) { max = scores[p]; winner = p; } }
        document.getElementById('high-score-display').innerText = `🏆 المتصدر: ${winner} (${max})`;
    }

    function resetScoresAndPlay() {
        players.forEach(p => scores[p] = 0);
        updateHighscore();
        goToCategorySelection();
    }

    function showScreen(id) {
        document.querySelectorAll('.container > div:not(.high-score-box)').forEach(d => d.classList.add('hidden'));
        document.getElementById(id).classList.remove('hidden');
    }

    function resetQuestions() { qIdx = 0; cycleQuestions(); document.getElementById('q-options').classList.add('hidden'); document.getElementById('q-next-btn').classList.remove('hidden'); }
</script>
